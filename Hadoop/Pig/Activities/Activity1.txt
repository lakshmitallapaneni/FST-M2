#create txt and csv file
$ vim txtFile.txt
$ vim csvFile.csv

#copy file to hdfs
$ hdfs dfs -put ./txtFile.txt /user/lakshmi
$ hdfs dfs -put ./csvFile.csv /user/lakshmi

#View the content of the file in hdfs
$ hdfs dfs -cat /user/lakshmi/txtFile.txt
$ hdfs dfs -cat /user/lakshmi/csvFile.csv

$ vim txtFile.txt

$ vim csvFile.csv

# Make a textData folder on the HDFS
$ hdfs dfs -mkdir /user/lakshmi/textData

# Copy the TXT file, txtFile.txt, into the textData folder in the HDFS
$ hdfs dfs -put ./txtFile.txt /user/lakshmi/textData

# Copy the CSV file, csvFile.csv, into the textData folder in the HDFS
$ hdfs dfs -put ./csvFile.csv /user/lakshmi/textData

# Use the stat command to get the file statistics
$ hdfs dfs -stat "type:%F permissions:%a %u:%g size:%b name:%n" /user/lakshmi/textData/txtFile.txt
$ hdfs dfs -stat "type:%F permissions:%a %u:%g size:%b name:%n" /user/lakshmi/textData/csvFile.csv
